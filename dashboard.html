⚙️ Código para tu Apps Script
Agrega estos 3 casos en el switch/if de tu función doPost en el Apps Script para que el dashboard pueda crear, borrar y pausar mensajes: 
case 'agregar_mensaje_programado': {
  var s = ss.getSheetByName('MENSAJES_PROGRAMADOS')
         || ss.insertSheet('MENSAJES_PROGRAMADOS');
  if (s.getLastRow() === 0)
    s.appendRow(['ID','Texto','Intervalo_min','Destino','Activo','Ultimo_envio','Creado']);
  s.appendRow([data.id, data.texto, data.intervalo_min,
               data.destino, 'si', '', data.creado]);
  return ok();
}
case 'borrar_mensaje_programado': {
  var s = ss.getSheetByName('MENSAJES_PROGRAMADOS');
  if (s) {
    var vals = s.getDataRange().getValues();
    for (var i = vals.length - 1; i >= 1; i--)
      if (vals[i][0].toString() === data.id.toString()) { s.deleteRow(i+1); break; }
  }
  return ok();
}
case 'toggle_mensaje_programado': {
  var s = ss.getSheetByName('MENSAJES_PROGRAMADOS');
  if (s) {
    var vals = s.getDataRange().getValues();
    for (var i = 1; i < vals.length; i++)
      if (vals[i][0].toString() === data.id.toString()) { s.getRange(i+1,5).setValue(data.activo); break; }
  }
  return ok();
}
Donde ss es tu SpreadsheetApp y ok() retorna JSON {ok:true}.
